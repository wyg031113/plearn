\section{etcd编译安装}
\begin{minted}[linenos,firstnumber=1, fontfamily=tt, baselinestretch=0.7, numberblanklines=false, style=vs,xleftmargin=16pt]{shell}
#etcd编译
mkdir -p  $GOPATH/src/go.etcd.io/
cd $GOPATH/src/go.etcd.io/
git clone https://github.com/etcd-io/etcd.git
./build
./bin/etcd
\end{minted}

 \subsection{画布操作}
 \begin{enumerate}
 	\item 新建画布 \tab Ctrl + N
 	\item 画布切换 \tab F
 	\item 复位工作区 \tab Alt -> W -> K -> R
 	\item 放大缩小 \tab Alt + 鼠标滚轮(Ctrl + +, Ctrl + -)
 	\item 缩放工具 \tab Z(放大：鼠标点击画布，或按下Alt点击画布，可以放大缩小)
 	\item 移动画布 \tab Space + 鼠标左键按下拖动
 	\item 切换画布 \tab Ctrl + Tab
 	\item 显示网格 \tab Ctrl + '
 	\item 显示参考线 \tab Ctrl + ;
 	\item 显示标尺 \tab Ctrl + R
 \end{enumerate}
